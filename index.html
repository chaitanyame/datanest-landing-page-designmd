aggregate.js
const orders = db.collection("orders"); const pipeline = [ { $match: { status: "shipped" }}, { $group: { _id: "$region", total: { $sum: "$amount" }, count: { $sum: 1 } }}, { $sort: { total: -1 }}, { $limit: 10 } ]; const results = await orders.aggregate(pipeline).toArray();
indexes.js
const users = db.collection("users"); await users.createIndex( { email: 1 }, { unique: true } ); await users.createIndex( { location: "2dsphere" } ); await users.createIndex( { bio: "text" }, { weights: { bio: 10 }} ); // Compound index for sorting await users.createIndex( { created: -1, status: 1 } );
